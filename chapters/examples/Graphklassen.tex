\chapter{Berechnungstheoretische Vorteile bestimmter Graphklassen zur Bestimmung bedingter Unabhängigkeit}
\label{ch:graphclasses}

Die folgenden Abschnitte orientieren sich an \textcite{handbook__of_formal_argumentation__vol__3:2024}.
Zu Beginn eine Feststellung, die die späteren komplexitätstheoretischen Einordnungen vereinfacht.
\begin{Lemma}
\label{lem:at-most-one-labelling-ci}
    Sei \(F=(A,R)\) ein AF und sei \(\sigma\) eine beliebige Semantik. 
    Falls \(|\Lambda_\sigma(F)| \leq 1\), dann ist die \(\sigma\)-Unabhängigkeit trivial.
\end{Lemma}

\begin{proof}
    Seien \(X,Y,Z \subseteq A\) paarweise disjunkt. Nach Definition gilt \(X \perp_\sigma Y \mid Z\), falls für alle \(\lambda_1,\lambda_2 \in \Lambda_\sigma(F)\) 
    mit \(\lambda_1|_Z = \lambda_2|_Z\) ein 
    \(\lambda_3 \in \Lambda_\sigma(F)\) existiert, sodass \(\lambda_3|_X = \lambda_1|_X\), \(\lambda_3|_Y = \lambda_2|_Y\) und \(\lambda_3|_Z = \lambda_1|_Z = \lambda_2|_Z\).
    Falls \(\Lambda_\sigma(F)=\emptyset\), ist die universelle Quantifikation über \(\lambda_1\) und \(\lambda_2\) leer. Die Aussage gilt daher vakuos.
    Sei nun \(\Lambda_\sigma(F)=\{\lambda\}\). Dann gilt für alle \(\lambda_1,\lambda_2 \in \Lambda_\sigma(F)\) bereits \(\lambda_1=\lambda_2=\lambda\). 
    Wähle \(\lambda_3=\lambda\). Dann gilt unmittelbar \(\lambda_3|_X=\lambda_1|_X\), \(\lambda_3|_Y=\lambda_2|_Y\) und \(\lambda_3|_Z=\lambda_1|_Z=\lambda_2|_Z\). 
    Also gilt \(X \perp_\sigma Y \mid Z\).
\end{proof}
\section{Azyklische AFs}
\label{sec:acyc-afs}

\begin{Proposition}
\label{prop:acyc-af-ci-st-top}
    \(\sigma\)-Unabhängigkeit für \(\sigma \in {co,st,pr}\) ist trivial in azyklischen AFs.
\end{Proposition}

\begin{proof}
\label{pr:acyc-af-ci-st-top-true}
    Sei \(F=(A,R)\) ein endlicher azyklischer AF. Für azyklische AFs fallen die \emph{complete}-, \emph{preferred}- und \emph{stable}-Semantik mit der \emph{grounded}-Semantik zusammen. 
    Da das grounded Labelling eindeutig ist, existiert genau ein Labelling \(\lambda\), sodass \(\Lambda_{co}(F)=\Lambda_{pr}(F)=\Lambda_{st}(F)=\{\lambda\}\).
    Sei nun \(\sigma \in \{co,st,pr\}\). Dann gilt insbesondere \(|\Lambda_\sigma(F)|=1\). Nach Lemma~\ref{lem:at-most-one-labelling-ci} folgt für alle paarweise disjunkten \(X,Y,Z \subseteq A\), 
    dass \(X \perp_\sigma Y \mid Z\) gilt.
\end{proof}

\section{Geradezykelfreie AFs}
\label{sec:even-cycle-free-afs}

Für AFs ohne gerade Zykel ist allein das \emph{grounded}-Labelling ein Kandidat, ein \emph{stable}-Labelling zu sein. 
Damit existiert ebenfalls maximal ein \emph{stable}-Labelling. 

\begin{Proposition}
\label{prop:even-cycle-free-af-ci-st-top}
    \(\sigma\)-Unabhängigkeit für \(\sigma \in {co,st,pr}\) ist trivial in geradezykelfreien AFs.
\end{Proposition}

\begin{proof}
\label{even-cycle-free-af-ci-top-true}
    Sei \(F=(A,R)\) ein AF ohne gerade Zyklen. Für diese Klasse fallen die \emph{complete}- und die \emph{preferred}-Semantik mit der \emph{grounded}-Semantik zusammen. 
    Da das grounded Labelling eindeutig ist, gilt \(|\Lambda_{co}(F)|= |\Lambda_{pr}(F)|=1\).
    Für die \emph{stable}-Semantik ist das grounded Labelling der einzige mögliche Kandidat. 
    Daher gilt entweder \(\Lambda_{st}(F)=\emptyset\), falls das \emph{grounded }Labelling nicht \emph{stable} ist, oder \(\Lambda_{st}(F)=\{\lambda\}\), 
    falls das \emph{grounded} Labelling \emph{stable} ist. In beiden Fällen gilt \(|\Lambda_{st}(F)| \leq 1\).
    Somit gilt für jedes \(\sigma \in \{co,st,pr\}\), dass \(|\Lambda_\sigma(F)| \leq 1\). 
    Nach Lemma~\ref{lem:at-most-one-labelling-ci} folgt für alle paarweise disjunkten \(X,Y,Z \subseteq A\), dass \(X \perp_\sigma Y \mid Z\) gilt.
\end{proof}


\section{Ungeradezykelfreie AFs}
\label{sec:odd-cycle-free-afs}

\begin{Proposition}
    \label{prop:ocf-af-ci-st-pi^2}
    \(stable\)-Unabhängigkeit ist auch in ungeradezykelfreien AFs \(\Pi_2^\text{P}\)-vollständig.
\end{Proposition}

\begin{proof}
    \label{pr:ocf-af-ci-st-pi^2-is-correct}
    Der Zugehörigkeitsbeweis erfolgt analog zum allgemeinen Fall.
    Für die Härte benutzen wir die Konstruktion von \textcite{bluemel:2025:independence-aba}, wobei zu zeigen bleibt, dass es sich um einen ungeradezykelfreien Graphen handelt.
    Wir betrachten zunächst die stark zusammenhängenden Komponenten von \(F_{\Psi}\). 
    Für jede universell quantifizierte Variable \(x_i\) bildet das zugehörige Argumentpaar 
    \(\{x_i,\overline{x_i}\}\) eine stark zusammenhängende Komponente. Entsprechend bildet 
    für jede existenziell quantifizierte Variable \(y_j\) das Argumentpaar 
    \(\{y_j,\overline{y_j}\}\) eine stark zusammenhängende Komponente. Darüber hinaus bildet 
    jedes Klauselargument \(c_k\) eine einelementige stark zusammenhängende Komponente, während 
    die Argumente \(\varphi\) und \(t\) gemeinsam die stark zusammenhängende Komponente 
    \(\{\varphi,t\}\) bilden.
    Somit gilt
       \( \operatorname{SCC}(F_{\Psi})
        =
        \bigl\{\{x_i,\overline{x_i}\} \mid x_i \in \{x_0\} \cup X\bigr\}
        \cup
        \bigl\{\{y_j,\overline{y_j}\} \mid y_j \in Y\bigr\}
        \cup
        \bigl\{\{c_k\} \mid 1 \leq k \leq m\bigr\}
        \cup
        \bigl\{\{\varphi,t\}\bigr\}.\)

    Zwischen den stark zusammenhängenden Komponenten verlaufen die Angriffe ausschließlich 
    in der Richtung Variablen nach Klauseln nach \(\{\varphi, t\}\). Der Kondensationsgraph von \(F_{\Psi}\) ist daher azyklisch.
    Gerichtete Zyklen können also nur in den oben beschriebenen Komponenten auftreten.
    Die einzigen nichttrivialen stark zusammenhängenden Komponenten sind die Argumentpaare
        \(x_i \leftrightarrow \overline{x_i},
        y_j \leftrightarrow \overline{y_j},
        \varphi \leftrightarrow t.\)
    Jede dieser Komponenten enthält ausschließlich einen gerichteten Zyklus der Länge zwei. 
    Die einelementigen Komponenten \(\{c_k\}\) enthalten keine Zyklen, da die Klauselargumente 
    sich nicht selbst angreifen.
    Damit besitzt jeder gerichtete Zyklus in \(F_{\Psi}\) gerade Länge. Folglich enthält 
    \(F_{\Psi}\) keinen gerichteten Zyklus ungerader Länge und ist somit ungeradezykelfrei.
\end{proof}

\begin{Proposition}
    \label{prop:ocf-af-ci-pr-pi^2-c}
    \(preferred\)-Unabhängigkeit ist auch in ungeradezykelfreien AFs \(\Pi_2^\text{P}\)-vollständig .
\end{Proposition}

\begin{proof}
    \label{pr:ocf-af-ci-pr-pi^2-c-is-correct}
    Folgt direkt aus Proposition \ref{prop:ocf-af-ci-st-pi^2} und der Tatsache, dass in ungeradezykelfreien AFs \(\Lambda_{st} = \Lambda_{pr}\).
\end{proof}

\begin{Proposition}
    \label{prop:ocf-af-ci-co-pi^2-c}
    \(complete\)-Unabhängigkeit ist auch in ungeradezykelfreien AFs \(\Pi_2^\text{P}\)-vollständig.
\end{Proposition}

\begin{proof}
    \label{pr:ocf-af-ci-co-pi^2-c-is-correct}
    Wir modifizieren die Konstruktion von \textcite{bluemel:2025:independence-aba}.
    Wir reduzieren vom \(\Pi_2^P\)-vollständigen Problem
    \(\mathrm{QBF}^{2}_{\forall}\). Eine Instanz dieses Problems ist
    eine quantifizierte boolesche Formel der Form
    \(\Psi=\forall X\exists Y\,\varphi(X,Y), \)
    wobei \(\varphi=\bigwedge_{k=1}^{m}C_k\) in konjunktiver
    Normalform vorliegt. Gefragt ist, ob für jede Belegung der
    Variablen aus \(X\) eine Belegung der Variablen aus \(Y\)
    existiert, welche \(\varphi\) erfüllt.


    \paragraph{Konstruktion.}
    Wir führen wir eine neue universelle
    Variable \(x_0\) ein und setzen
    \( X^{+}=X\cup\{x_0\}~
        \text{und}~
        \widehat{\varphi}
        =
        \bigwedge_{k=1}^{m}(x_0\lor C_k).\)
    Damit gilt
    \(
        \forall X\exists Y\,\varphi~
        \Longleftrightarrow~
        \forall X^{+}\exists Y\,\widehat{\varphi}.
    \)   
    Aus \(\Psi\) konstruieren wir das AF
    \(F_{\Psi}=(A_{\Psi},R_{\Psi})\) mit
    \[
    \begin{split}
        A_{\Psi}={}&
        \{x_i,\overline{x_i},d_i\mid x_i\in X^{+}\}
        \cup
        \{y_j,\overline{y_j}\mid y_j\in Y\}\\
        &{}\cup
        \{c_1,\ldots,c_m,\varphi,t,s,g\}.
    \end{split}
    \]
    Die Angriffsmenge ist definiert als:
    \[
    \begin{split}
        R_{\Psi}={}&
        \{(x_i,\overline{x_i}),(\overline{x_i},x_i)
            \mid x_i\in X^{+}\}\\
        &{}\cup
        \{(y_j,\overline{y_j}),(\overline{y_j},y_j)
            \mid y_j\in Y\}\\
        &{}\cup
        \{(\ell,c_k)
            \mid \ell\text{ kommt in der Klausel }
            x_0\lor C_k\text{ vor}\}\\
        &{}\cup
        \{(c_k,\varphi)\mid 1\leq k\leq m\}\\
        &{}\cup
        \{(\varphi,t),(t,\varphi)\}\\
        &{}\cup
        \{(x_i,d_i),(\overline{x_i},d_i),(d_i,s)
            \mid x_i\in X^{+}\}\\
        &{}\cup
        \{(s,g),(g,\varphi)\}\\
        &{}\cup
        \{(g,c_k)\mid 1\leq k\leq m\}.
    \end{split}
    \]
    Die Konstruktion ist offensichtlich polynomial in der
    Größe von \(\Psi\).
    Die Argumente \(d_i\) erkennen, ob das jeweilige Paar
    \(x_i,\overline{x_i}\) total gelabelt ist. Das Argument \(s\) erhält genau dann das Label \(in\), wenn alle
    universellen Variablen total gelabelt sind. Das Argument
    \(g\) deaktiviert im partiellen Fall die eigentliche
    Formelauswertung.
    Als Instanz der \(complete\)-Unabhängigkeit geben wir
    \(
        \{\varphi\}
        \perp_{co}
        X^{+}
        \mid
        \{s\}
    \)
    aus.  Im Folgenden zeigen wir:
    \(
        \Psi\text{ ist gültig}
    \)
    genau dann, wenn
    \(
        \{\varphi\}
        \perp_{co}
        X^{+}
        \mid
        \{s\}
        \text{ in }F_{\Psi}.
    \)
    Zunächst gilt in jedem \(complete\)-Labelling \(\lambda\):
    \(
        \lambda(s)=in
    \)
    genau dann, wenn alle Variablen aus \(X^{+}\) total gelabelt sind.
    In diesem Fall gilt \(\lambda(g)=out\). Ist dagegen mindestens
    eine Variable aus \(X^{+}\) unentschieden, so gilt
    \(\lambda(s)=\lambda(g)=und\).

    \begin{figure}[H]
        \myfloatalign
        \centering
        \includegraphics[keepaspectratio, scale=0.8]{../../Bilder/ocf_co_reduction_diagram(1).pdf}
        \caption[Reduktion]{Konstruktionsbeispiel Reduktion}
        \label{fig:ocf-co-reduction}
    \end{figure}

    \paragraph{Korrektheit.}
    Sei zunächst \(\Psi\) gültig und seien
    \(\lambda_1,\lambda_2\in\Lambda_{co}(F_{\Psi})\) mit
    \(\lambda_1(s)=\lambda_2(s)\). Wir konstruieren ein vollständiges
    Labelling \(\lambda_3\), das das Label von \(\varphi\) aus
    \(\lambda_1\) und das Labelling von \(X^{+}\) aus \(\lambda_2\)
    übernimmt.
    Falls \(\lambda_1(s)=\lambda_2(s)=und\), gilt \(g=und\).
    Daher kann \(\varphi\) nicht mit \(in\), sondern nur mit \(out\)
    oder \(und\) gelabelt sein. Beide Labels sind unabhängig vom
    konkreten Labelling von \(X^{+}\) realisierbar: Für
    \(\varphi=out\) wählen wir \(t=in\), für \(\varphi=und\) wählen
    wir \(t=und\).
    Falls \(\lambda_1(s)=\lambda_2(s)=in\), kodiert
    \(\lambda_2|_{X^{+}}\) eine totale Belegung \(\alpha\) von
    \(X^{+}\), und es gilt \(g=out\). Wegen der Gültigkeit von
    \(\forall X^{+}\exists Y\,\widehat{\varphi}\) existiert eine
    Belegung \(\beta\) von \(Y\) mit
    \(\widehat{\varphi}(\alpha,\beta)=\top\). Wir labeln die
    \(Y\)-Argumente gemäß \(\beta\). Dadurch wird jedes
    Klauselargument von einem \(in\)-gelabelten Literal angegriffen
    und somit mit \(out\) gelabelt. Daher können die drei möglichen
    Labels von \(\varphi\) durch
    \(
        (\varphi,t)\in
        \{(in,out),(out,in),(und,und)\}
    \)
    realisiert werden. Somit existiert in beiden Fällen das
    erforderliche Labelling \(\lambda_3\), und es gilt
    \(\{\varphi\}\perp_{co}X^{+}\mid\{s\}\).

    Sei nun \(\Psi\) nicht gültig. Dann existiert eine Belegung
    \(\alpha\) von \(X\), sodass
    \(
        \forall\beta:Y\to\{\top,\bot\}\colon
        \varphi(\alpha,\beta)=\bot.
    \)
    Erweitere \(\alpha\) durch \(x_0=\bot\) zu einer totalen Belegung
    \(\alpha^{+}\) von \(X^{+}\). Wir erhalten ein \emph{complete}-Labelling \(\lambda_2\).
    Separat existiert ein \emph{complete}-Labelling \(\lambda_1\) mit
    \(\lambda_1(s)=in\) und \(\lambda_1(\varphi)=in\): Dazu wird
    \(x_0=in\) gewählt, sodass \(x_0\) alle Klauselargumente angreift
    und diese mit \(out\) gelabelt werden.
    Damit sind \(\lambda_1~\text{und}~\lambda_2\) realisierbare \emph{complete}-Labellings, wohingegen ein 
    kombinierendes Labelling \(\lambda_3\) nicht existieren kann. Denn in \(\lambda_2\) existiert immer mindestens ein
    \(c_k\), das mit \(in\) gelabelt ist und in \(\lambda_1\) trägt \(\varphi\) das \(in\) Label. Dies ist ein Widerspruch, da 
    \(c_k ~\varphi\) angreift. 
    \paragraph{Graphklassenzugehörigkeit}
    Das konstruierte AF \(F_{\Psi}\) ist ungeradezykelfrei. 
    Seine einzigen nichttrivialen stark zusammenhängenden Komponenten sind
    \(\{x_i,\overline{x_i}\}\) für \(x_i\in X^{+}\),
    \(\{y_j,\overline{y_j}\}\) für \(y_j\in Y\) sowie
    \(\{\varphi,t\}\).
    Die Argumente \(d_i\), \(s\), \(g\) und \(c_k\) bilden jeweils
    einelementige stark zusammenhängende Komponenten.
    Zwischen diesen Komponenten verlaufen die Angriffe ausschließlich in den
    Richtungen
    \(\{x_i,\overline{x_i}\}\to d_i\to s\to g\),
    \(\{x_i,\overline{x_i}\},\{y_j,\overline{y_j}\}\to c_k\),
    \(g\to c_k\) und
    \(g,c_k\to\{\varphi,t\}\).
    Der Kondensationsgraph von \(F_{\Psi}\) ist daher azyklisch.
    Folglich liegt jeder gerichtete Zyklus vollständig in einer der
    nichttrivialen Komponenten. Diese enthalten jeweils ausschließlich den
    Zweierzyklus
    \(x_i\leftrightarrow\overline{x_i}\),
    \(y_j\leftrightarrow\overline{y_j}\) beziehungsweise
    \(\varphi\leftrightarrow t\).
    Somit besitzt jeder gerichtete Zyklus gerade Länge und
    \(F_{\Psi}\) ist ungeradezykelfrei.
\end{proof}

\section{Bipartite AFs}
\begin{Proposition}
\label{prop:bip-st-ind-hard}
    \(stable\)-Unabhängigkeit ist auch in bipartiten AFs \(\Pi_2^\text{P}\)-vollständig.
\end{Proposition}

\begin{proof}
    Wir reduzieren vom \(\Pi_2^P\)-vollständigen Problem
    \textsc{Balanced Monotone \(\forall\exists\)-3-SAT-\((1,1,2,2)\)}.

    \paragraph{Konstruktion.}
Sei
\(
    \Psi=\forall X\exists Y\,\varphi
\)
eine Instanz von \textsc{Balanced Monotone \(\forall\exists\)-3-SAT-\((1,1,2,2)\)}.
Die Klauselmenge von \(\varphi\) zerfällt in die positiven Klauseln
\(\mathcal{C}^{+}\) und die negativen Klauseln \(\mathcal{C}^{-}\).
Wir konstruieren das AF
\(
    F_{\Psi}=(\mathcal{A}_{\Psi},R_{\Psi}).
\)
Die Argumentmenge ist gegeben durch
\[
\begin{aligned}
    \mathcal{A}_{\Psi}
    ={}&
    \{v,\overline{v}\mid v\in X\cup Y\cup\{p,n\}\}\\
    &\cup
    \{c_C\mid C\in\mathcal{C}^{+}\cup\mathcal{C}^{-}\}\\
    &\cup
    \{\varphi^{+},\varphi^{-},t^{+},t^{-}\},
\end{aligned}
\]
wobei \(p\) und \(n\) neu eingeführte Variablen sind.
Die Angriffsrelation ist
\[
\begin{aligned}
    R_{\Psi}
    ={}&
    \{(v,\overline{v}),(\overline{v},v)
        \mid v\in X\cup Y\cup\{p,n\}\}\\
    &\cup
    \{(v,c_C)
        \mid C\in\mathcal{C}^{+},\,v\in C\}\\
    &\cup
    \{(\overline{v},c_C)
        \mid C\in\mathcal{C}^{-},\,\neg v\in C\}\\
    &\cup
    \{(p,c_C)\mid C\in\mathcal{C}^{+}\}\\
    &\cup
    \{(\overline{n},c_C)\mid C\in\mathcal{C}^{-}\}\\
    &\cup
    \{(c_C,\varphi^{+})\mid C\in\mathcal{C}^{+}\}\\
    &\cup
    \{(c_C,\varphi^{-})\mid C\in\mathcal{C}^{-}\}\\
    &\cup
    \{(\varphi^{+},t^{+}),(t^{+},\varphi^{+})\}\\
    &\cup
    \{(\varphi^{-},t^{-}),(t^{-},\varphi^{-})\}.
\end{aligned}
\]
Als \(stable\)-Unabhängigkeitsinstanz wählen wir
\(\{\varphi^{+},\varphi^{-}\} \perp_{st} X \cup \{p,n\} \mid \emptyset\).

Die gegenseitigen Angriffe
\(
    v\leftrightarrow\overline{v}
\)
erzwingen unter stabiler Semantik, dass genau eines der beiden
Literalargumente \(in\) ist. Jedes stabile Labelling kodiert damit eine
totale Wahrheitsbelegung.
Ein positives Klauselargument \(c_C\) wird von den positiven
Literalargumenten seiner Klausel sowie von \(p\) angegriffen. Ein negatives
Klauselargument wird von den entsprechenden negativen Literalargumenten
sowie von \(\overline{n}\) angegriffen. Wenn
\(
    p=out
    \text{und}
    n=in,
\)
dann werden die Klauselargumente ausschließlich durch die ursprünglichen Literale
kontrolliert werden.
Falls
\(
    p=in
    \text{und}
    n=out
\)
sind \(p\) und \(\overline{n}\) jeweils \(in\). Dadurch werden alle positiven
beziehungsweise negativen Klauselargumente angegriffen. Somit ist
\(
    \varphi^{+}=\varphi^{-}=in
\)
unabhängig von der ursprünglichen Formel separat realisierbar.
Die Argumente \(t^{+}\) und \(t^{-}\) erlauben es, die Labels von
\(\varphi^{+}\) und \(\varphi^{-}\) unabhängig auf \(in\) oder \(out\) zu
setzen, sofern die zugehörigen Klauselargumente \(out\) sind.

\paragraph{Graphklassenzugehörigkeit.}
Definiere
\[
\begin{aligned}
    L={}&
    \{v\mid v\in X\cup Y\cup\{p,n\}\}\\
    &\cup
    \{c_C\mid C\in\mathcal{C}^{-}\}
    \cup
    \{\varphi^{+},t^{-}\},
\end{aligned}
\]
und
\[
\begin{aligned}
    R={}&
    \{\overline{v}\mid v\in X\cup Y\cup\{p,n\}\}\\
    &\cup
    \{c_C\mid C\in\mathcal{C}^{+}\}
    \cup
    \{\varphi^{-},t^{+}\}.
\end{aligned}
\]
Jeder Angriff in \(R_{\Psi}\) verläuft zwischen \(L\) und \(R\). Somit gilt
\(
    R_{\Psi}\subseteq(L\times R)\cup(R\times L),
\)
und \(F_{\Psi}\) ist bipartit.

    \begin{figure}[htbp]
        \myfloatalign
        \centering
        \includegraphics[width=0.9\textwidth]{../../Bilder/bipartite_AF-reduction.pdf}
        \caption[Reduktion]{Konstruktionsbeispiel Reduktion}
        \label{fig:bipartite-st-reduction}
    \end{figure}

    \paragraph{Korrektheit.}
    In jedem stabilen Labelling ist aus jedem Paar \(v,\overline{v}\) genau ein Argument
    mit \(in\) gelabelt. Die Einschränkung eines stabilen Labellings auf die
    Variablenargumente entspricht daher einer totalen Wahrheitsbelegung.

    Angenommen, \(\Psi\) ist gültig. Seien \(\lambda_1\) und \(\lambda_2\) beliebige
    stabile Labellings. Das Labelling \(\lambda_1\) legt die gewünschten Labels von
    \(\varphi^{+}\) und \(\varphi^{-}\) fest, während \(\lambda_2\) eine Belegung von
    \(X\cup\{p,n\}\) festlegt. Da \(\Psi\) gültig ist, existiert zu der durch
    \(\lambda_2\) bestimmten \(X\)-Belegung eine Belegung von \(Y\), welche alle
    ursprünglichen Klauseln von \(\varphi\) erfüllt. Diese Belegung erfüllt die
    Klauseln unabhängig von den Werten der Variablen \(p\) und \(n\). Daher werden
    sämtliche Klauselargumente mit \(out\) gelabelt.

    Anschließend können die Labels von \(\varphi^{+}\) und \(\varphi^{-}\) unabhängig
    gewählt werden: Für \(\varphi^{+}=in\) wird \(t^{+}=out\) gesetzt, während für
    \(\varphi^{+}=out\) das Argument \(t^{+}\) mit \(in\) gelabelt wird. Analog wird
    mit \(\varphi^{-}\) und \(t^{-}\) verfahren. Somit existiert ein stabiles Labelling,
    das die Einschränkung von \(\lambda_1\) auf \(A\) und die Einschränkung von
    \(\lambda_2\) auf \(B\) kombiniert.

    Sei nun \(\Psi\) nicht gültig. Dann existiert eine Belegung
    \(\alpha\) von \(X\), sodass keine Belegung von \(Y\) die Formel \(\varphi\)
    unter \(\alpha\) erfüllt. Ergänze diese Belegung durch \(p=\perp\) und \(n=\top\).
    Im zugehörigen stabilen Labelling \(\lambda_1\) sind \(p\) und \(\overline{n}\) mit \(out\)
    gelabelt.

    Andererseits ist das Labelling
    \(\varphi^{+}=in\) und \(\varphi^{-}=in\) als \(\lambda_2\) realisierbar. Hierzu wird
    \(p=\top\) und \(n=\perp\) gewählt. Dann sind \(p\) und \(\overline{n}\) mit
    \(in\) gelabelt und greifen alle positiven beziehungsweise negativen
    Klauselargumente an. Daher sind sämtliche Klauselargumente \(out\), sodass
    \(\varphi^{+}\) und \(\varphi^{-}\) gleichzeitig \(in\) sein können.

    Ein kombinierendes stabiles Labelling müsste gleichzeitig die nicht erfüllende
    Belegung \(\alpha\), die Labels \(p=out,n=in\) und
    \(\varphi^{+}=\varphi^{-}=in\) realisieren. Aus
    \(\varphi^{+}=\varphi^{-}=in\) folgt, dass alle positiven und negativen
    Klauselargumente \(out\) sind. Da \(p\text{ und } \overline{n}\) keine Klauselargumente
    angreifen, muss jede Klausel von mindestens einem \(in\)-gelabelten ursprünglichen
    Literalargument angegriffen werden. Die dadurch bestimmte Belegung von \(Y\) würde
    \(\varphi\) unter \(\alpha\) erfüllen, im Widerspruch zur Wahl von \(\alpha\).
\end{proof}

\begin{Proposition}
    \label{prop:bip-prf-ind-hard}
    \(preferred\)-Unabhängigkeit ist auch in bipartiten AFs \(\Pi_2^\text{P}\)-vollständig.
\end{Proposition}

\begin{proof}
    Folgt direkt aus Proposition \ref{prop:bip-st-ind-hard} und der Tatsache, dass 
    bipartite Graphen eine Unterklasse der ungeradezykelfreien Graphen sind. Womit direkt
    \(\Lambda_{st} = \Lambda_{pr}\) gilt. 
\end{proof}

\section{Symmetrische AFs}
\label{sec:sym-afs}

In diesem Abschnitt betrachten wir Argumentationssysteme mit besonders hoher
Angriffsdichte. Die zugrunde liegende Beobachtung ist, dass zusätzliche Angriffe
die Kombinierbarkeit von Labellings einschränken können. Dadurch treten
Verletzungen bedingter Unabhängigkeit häufiger auf. Als Extremfall betrachten
wir zunächst vollständig gerichtete Argumentationssysteme.

\subsection{Vollständig gerichtete Graphen}
\label{subsec:cdg-afs}

Wir nennen ein AF \(F=(A,R)\) \emph{vollständig gerichtet}, wenn für alle
verschiedenen Argumente \(a,b\in A\) sowohl \((a,b)\in R\) als auch
\((b,a)\in R\) gilt. Jedes Argument greift also jedes andere Argument an.

Das folgende Lemma beschreibt die Struktur der \textit{stable}-Labellings in
vollständig gerichteten AFs.

\begin{Lemma}
    \label{lem:cdg-afs-labellings}
    Sei \(F=(A,R)\) ein vollständig gerichtetes AF mit \(A\neq\emptyset\).
    Dann gilt
    \[
        |\Lambda_{\mathrm{st}}(F)|=|A|.
    \]
\end{Lemma}

\begin{proof}
    Für jedes Argument \(a\in A\) definieren wir ein Labelling \(\lambda_a\)
    durch
    \[
        \mathit{in}(\lambda_a)=\{a\}
        \qquad\text{und}\qquad
        \mathit{out}(\lambda_a)=A\setminus\{a\}.
    \]
    Da \(a\) jedes Argument aus \(A\setminus\{a\}\) angreift, ist jedes
    Argument mit Label \(\mathit{out}\) von einem Argument mit Label
    \(\mathit{in}\) angegriffen. Zugleich wird \(a\) von keinem Argument mit
    Label \(\mathit{in}\) angegriffen, da \(a\) das einzige Argument mit Label
    \(\mathit{in}\) ist. Somit ist \(\lambda_a\) ein
    \textit{stable}-Labelling von \(F\). Da jedes \(a\in A\) ein solches
    Labelling liefert, folgt
    \[
        |\Lambda_{\mathrm{st}}(F)|\geq |A|.
    \]

    Umgekehrt sei \(\lambda\in\Lambda_{\mathrm{st}}(F)\). Da
    \(\lambda\) ein \textit{stable}-Labelling ist, gilt
    \(\mathit{und}(\lambda)=\emptyset\). Außerdem kann
    \(\mathit{in}(\lambda)\) nicht leer sein: Wäre
    \(\mathit{in}(\lambda)=\emptyset\), so hätte kein Argument mit Label
    \(\mathit{out}\) einen Angreifer mit Label \(\mathit{in}\), im Widerspruch
    zur \textit{stable}-Bedingung.

    Es bleibt zu zeigen, dass höchstens ein Argument mit Label
    \(\mathit{in}\) existieren kann. Angenommen, es gäbe zwei verschiedene
    Argumente \(a,b\in A\) mit
    \[
        \lambda(a)=\mathit{in}
        \qquad\text{und}\qquad
        \lambda(b)=\mathit{in}.
    \]
    Da \(F\) vollständig gerichtet ist, gilt \((a,b)\in R\). Damit wird ein
    Argument mit Label \(\mathit{in}\) von einem Argument mit Label
    \(\mathit{in}\) angegriffen. Dies widerspricht der Bedingung, dass ein
    Argument genau dann \(\mathit{in}\) ist, wenn alle seine Angreifer
    \(\mathit{out}\) sind. Also enthält jedes \textit{stable}-Labelling genau
    ein Argument mit Label \(\mathit{in}\).

    Folglich sind die \textit{stable}-Labellings von \(F\) genau die
    Labellings \(\lambda_a\) mit \(a\in A\). Daher gilt
    \[
        |\Lambda_{\mathrm{st}}(F)|=|A|.
    \]
\end{proof}

Insbesondere existiert für jedes \(a\in A\) genau ein
\textit{stable}-Labelling \(\lambda_a\) mit
\[
    \mathit{in}(\lambda_a)=\{a\}.
\]
Damit kann jedes einzelne Argument in einem \textit{stable}-Labelling
akzeptiert werden, aber niemals gemeinsam mit einem anderen Argument.

\begin{Proposition}
    \label{prop:cdg-afs-not-st-ci}
    Sei \(F=(A,R)\) ein vollständig gerichtetes AF mit \(|A|\geq 2\).
    Seien \(X,Y,Z\subseteq A\) paarweise disjunkt mit
    \(X\neq\emptyset\) und \(Y\neq\emptyset\). Dann gilt
    \[
        X\not\bot_{\mathrm{st}}Y\mid Z.
    \]
\end{Proposition}

\begin{proof}
    Da \(X\) und \(Y\) nichtleer sind, wählen wir Argumente
    \(x\in X\) und \(y\in Y\). Wegen der paarweisen Disjunktheit gilt
    \(x\neq y\) sowie \(x,y\notin Z\).

    Nach Lemma~\ref{lem:cdg-afs-labellings} existieren
    \textit{stable}-Labellings \(\lambda_x,\lambda_y\in
    \Lambda_{\mathrm{st}}(F)\) mit
    \[
        \mathit{in}(\lambda_x)=\{x\}
        \qquad\text{und}\qquad
        \mathit{in}(\lambda_y)=\{y\}.
    \]
    Da weder \(x\) noch \(y\) in \(Z\) liegt, erhalten alle Argumente aus
    \(Z\) in beiden Labellings das Label \(\mathit{out}\). Also gilt
    \[
        \lambda_x|_Z=\lambda_y|_Z.
    \]

    Angenommen, \(X\bot_{\mathrm{st}}Y\mid Z\) würde gelten. Dann müsste es
    nach Definition der bedingten Unabhängigkeit ein
    \(\lambda_3\in\Lambda_{\mathrm{st}}(F)\) geben mit
    \[
        \lambda_3|_X=\lambda_x|_X,
        \qquad
        \lambda_3|_Y=\lambda_y|_Y,
        \qquad
        \lambda_3|_Z=\lambda_x|_Z.
    \]
    Insbesondere folgt daraus
    \[
        \lambda_3(x)=\mathit{in}
        \qquad\text{und}\qquad
        \lambda_3(y)=\mathit{in}.
    \]
    Dies ergibt allerdings sofort einen Widerspruch, da \((a,b) \in R\) ist und somit ein mit dem \(\mathit{in}\)-\emph{Label} versehenes Argument ein weiteres mit
    dem \(\mathit{in}\)-\emph{Label} versehenes Argument angreifen würde.  

    Also kann kein solches \(\lambda_3\) existieren. Damit gilt
    \[
        X\not\bot_{\mathrm{st}}Y\mid Z.
    \]
\end{proof}
